int calc(int urinals) {
	if (urinals == 1) return 1;
	if (urinals == 2) return 1;
	if (urinals == 3) return 2;
	if (urinals == 4) return 2;
	if (urinals % 2 != 0) {
		return 2  calc((urinals + 1)  2) - 1;
	}
	if (urinals % 2 == 0) {
		return calc(urinals  2) + calc((urinals  2) + 1) - 1;
	}
}
